Visit gethinode.com/license for more details. */}} {{- $error := false -}} {{/* Validate and initialize arguments using the icon structure. The icon structure only declares two positions (icon, class); this shortcode joins any further positional arguments into the class itself (see below), so only the declared positions are forwarded into validation to avoid excess-positional warnings on calls like `{{< fa "fire" "fa-stack-1x" "fa-inverse" >}}`. */}} {{- $validationArgs := .Params -}} {{- if not .IsNamedParams -}} {{- $validationArgs = first 2 .Params -}} {{- end -}} {{ $result := partial "utilities/Args.html" (dict "structure" "icon" "args" $validationArgs "named" .IsNamedParams "group" "shortcode" "strict" false) }} {{ $args := $result.args }} {{ if or $result.err $result.warnmsg }} {{ partial (cond $result.err "utilities/LogErr.html" "utilities/LogWarn.html") (dict "partial" "shortcodes/fa.html" "warnid" "warn-invalid-arguments" "msg" "Invalid arguments" "details" ($result.errmsg | append $result.warnmsg) "file" .Page.File "position" .Position )}} {{ end }} {{- $icon := $args.icon -}} {{- $class := $args.class -}} {{/* For positional args with 2+ args, join all args after icon as modifiers */}} {{- if and (not .IsNamedParams) (ge (len .Params) 2) -}} {{- $class = delimit (after 1 .Params) " " -}} {{- end -}} {{- $fullIcon := "" -}} {{- if and $icon $class -}} {{- $fullIcon = printf "far %s %s" $icon $class -}} {{- else if $icon -}} {{- $fullIcon = printf "far %s" $icon -}} {{- end -}} {{/* The "spacing" argument always carries a schema default of true, so it is never nil by the time it reaches here. Only fall back to the fa-li-aware default when the caller genuinely left it unset (per $result.defaulted); otherwise an explicit `spacing=false` would be silently discarded by a plain `| default` fallback. */}} {{- $spacing := $args.spacing -}} {{- if in $result.defaulted "spacing" -}} {{- $spacing = not (in (split (default "" $args.wrapper) " ") "fa-li") -}} {{- end -}} {{- if not $fullIcon -}} {{ partial "utilities/LogErr.html" (dict "partial" "shortcodes/fa.html" "msg" "Missing icon name" "file" .Page.File "position" .Position )}} {{ $error = true }} {{- end -}} {{- if not $error -}} {{- partial "assets/icon.html" (dict "page" .Page "icon" $fullIcon "inline-style" (or $args.inlineStyle $args.style) "wrapper" $args.wrapper "spacing" $spacing ) -}} {{- end -}}
